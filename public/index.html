Apni website me kaise lagaye

Copy ye code apne HTML me:

<form id="uploadForm">
  <input type="file" id="imageInput" accept="image/*">
  <button type="submit">Upload Photo</button>
</form>
<div id="result"></div>

<script>
document.getElementById('uploadForm').onsubmit = async (e) => {
  e.preventDefault()
  const file = imageInput.files[0]
  if (!file) return

  const formData = new FormData()
  formData.append('image', file)

  const res = await fetch('https://YOUR_BACKEND_URL/api/upload', {
    method: 'POST',
    body: formData
  })
  const data = await res.json()

  if (data.success) {
    // ✅ Ye URL database me save karo
    console.log('CDN URL:', data.url)
    result.innerHTML = '<img src="' + data.url + '">'
  }
}
</script>
